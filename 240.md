## PE 240

划分数搜一波然后加点剪枝

``` c++
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=155;
int n=20,m=12,k=10,ed=70;
ll a[N],ans;
inline ll fac(int x){
	ll res=1;
	for (ll i=2;i<=x;++i) res*=i;
	return res;
}
inline void chk(){
	ll cnt=fac(n);
	for (int i=1;i<=m;++i)
		cnt/=fac(a[i]);
	ans+=cnt;
}
void dfs(int x,int cnt,int s){
	if (!x){
		if (cnt==n) chk();
		return;
	}
	for (int i=0;i+cnt<=n;++i){
		if (cnt<k && cnt+i>=k && s+((k-cnt)*x)!=ed) continue;
		a[x]=i;
		dfs(x-1,cnt+i,s+(x*i));
	}
}
int main(){
	dfs(m,0,0);
	printf("%lld\n",ans);
	return 0;
}
```
